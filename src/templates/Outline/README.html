Outline

A fast, collaborative, knowledge base for your team built using React and Node.js.

Sources:

  • GitHub
  • Website
  • Selfhosting Documentation

Stack

### The compose file:

services:
  outline:
    container_name: outline
    image: docker.getoutline.com/outlinewiki/outline:latest
    ports:
      - "3000:3000"
    volumes:
      - ./outline/storage-data:/var/lib/outline/data
    depends_on:
      - postgres
      - redis
      - api
    environment:
      - DATABASE_URL="postgres://outlinedb:${POSTGRES_PASSWORD}@${local_ip}:${POSTGRES_PORT}/outline"
      - DATABASE_CONNECTION_POOL_MIN="1"
      - DATABASE_CONNECTION_POOL_MAX="8"
      - PGSSLMODE="disable"
      - REDIS_URL="redis://${local_ip}:${REDIS_PORT}"
      - FILE_STORAGE="local"
      - FILE_STORAGE_LOCAL_ROOT_DIR="/var/lib/outline/data"
      - FILE_STORAGE_UPLOAD_MAX_SIZE="262144000"
      - FILE_STORAGE_IMPORT_MAX_SIZE=""
      - FILE_STORAGE_WORKSPACE_IMPORT_MAX_SIZE=""
      - AWS_ACCESS_KEY_ID=""
      - AWS_SECRET_ACCESS_KEY=""
      - AWS_REGION=""
      - AWS_S3_ACCELERATE_URL=""
      - AWS_S3_UPLOAD_BUCKET_URL=""
      - AWS_S3_UPLOAD_BUCKET_NAME=""
      - AWS_S3_FORCE_PATH_STYLE=""
      - AWS_S3_ACL="public-read"
      - SLACK_CLIENT_ID=""
      - SLACK_CLIENT_SECRET=""
      - SLACK_VERIFICATION_TOKEN=""
      - SLACK_APP_ID=""
      - SLACK_MESSAGE_ACTIONS=""
      - GOOGLE_CLIENT_ID=""
      - GOOGLE_CLIENT_SECRET=""
      - AZURE_CLIENT_ID=""
      - AZURE_CLIENT_SECRET=""
      - AZURE_RESOURCE_APP_ID=""
      - OIDC_USERNAME_CLAIM="preferred_username"
      - OIDC_DISPLAY_NAME="OpenID Connect"
      - OIDC_SCOPES="openid profile email"
      - GITHUB_CLIENT_ID=""
      - GITHUB_CLIENT_SECRET=""
      - GITHUB_APP_NAME=""
      - GITHUB_APP_ID=""
      - GITHUB_APP_PRIVATE_KEY=""
      - SSL_KEY=""
      - SSL_CERT=""
      - FORCE_HTTPS="false"
      - SECRET_KEY="$SECRET_KEY"
      - UTILS_SECRET="$UTILS_SECRET"
      - NODE_ENV="production"
      - URL=""
      - PORT="3000"
      - COLLABORATION_URL=""
      - ENABLE_UPDATES="false"
      - DEBUG="http"
      - DEFAULT_LANGUAGE="en_US"
      - LOG_LEVEL="info"
      - SENTRY_DSN=""
      - SENTRY_TUNNEL=""
      - RATE_LIMITER_ENABLED="false"
      - RATE_LIMITER_REQUESTS="1000"
      - RATE_LIMITER_DURATION_WINDOW="10"
      - IFRAMELY_URL="${local_ip}:${IFRAMELY_PORT}"
      - IFRAMELY_API_KEY=""
      - OIDC_CLIENT_ID=""
      - OIDC_CLIENT_SECRET=""
      - OIDC_AUTH_URI=""
      - OIDC_TOKEN_URI=""
      - OIDC_USERINFO_URI=""
      - OIDC_LOGOUT_URI=""
      - SMTP_HOST=""
      - SMTP_PORT=""
      - SMTP_USERNAME=""
      - SMTP_PASSWORD=""
      - SMTP_FROM_EMAIL=""
      - SMTP_REPLY_EMAIL=""
      - SMTP_TLS_CIPHERS=""
      - SMTP_SECURE="false"

  redis:
    container_name: redis-outline
    image: redis
    volumes:
      - ./outline/redis.conf:/redis.conf
    ports:
      - "6379:6379"
    command: ["redis-server", "/redis.conf"]
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 30s
      retries: 3

  postgres:
    container_name: postgres-outline
    restart: always
    image: postgres
    ports:
      - "5432:5432"
    volumes:
      - ./outline/database-data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD", "pg_isready", "-d", "outline", "-U", "user"]
      interval: 30s
      timeout: 20s
      retries: 3
    environment:
      - POSTGRES_PASSWORD: "${POSTGRES_PASSWORD}"
      - POSTGRES_USER: "outlinedb"
      - POSTGRES_DB: "outline"


  api:
    image: ghcr.io/itteco/iframely:latest
    container_name: iframely-outline
    volumes:
      - ./outline/iframely/config.local.js:/iframely/config.local.js
    restart: always
    ports:
      - "8061:8061"
